type:vertex

#version 330 core

in vec3 position;
in vec2 texCoordsIn;
in mat4 mvpMatrix;

out vec2 texCoords;

void main() {
    texCoords = texCoordsIn;
    gl_Position = mvpMatrix * vec4(position, 1.0);
}

type:fragment

#version 330 core

in vec2 texCoords;

out vec4 outColor;

uniform sampler2D sampler;

void main() {
    outColor = texture(sampler, texCoords);
    outColor = vec4(1);
}
